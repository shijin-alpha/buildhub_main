File Upload Handler

const handleFileUpload = async (field, files) => { if (!files || files.length === 0) return; setUploadingFiles(true); try { const uploadedFiles = []; for (let file of files) { const previewUrl = URL.createObjectURL(file); uploadedFiles.push({ file: file, name: file.name, size: file.size, type: file.type, preview: previewUrl, uploaded: false }); } // Handle single vs multiple files if (field === 'layout_image') { setTechnicalDetails(prev => ({ ...prev, [field]: uploadedFiles[0] })); } else { setTechnicalDetails(prev => ({ ...prev, [field]: [...(prev[field] || []), ...uploadedFiles] })); } } catch (error) { console.error('Error handling file upload:', error); } finally { setUploadingFiles(false); } };
